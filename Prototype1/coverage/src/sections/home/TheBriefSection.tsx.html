function Reveal({ inView, delay = 0, children }: { inView: boolean; delay?: number; children: React.ReactNode }) {
    return (
        <motion.div initial={{ y: 16, opacity: 0 }} animate={{ y: inView ? 0 : 16, opacity: inView ? 1 : 0 }} transition={{ duration: 0.55, delay }}>
            {children}
        </motion.div>
    );
}
